14. 💛filter()

const newArray = 배열.filter(function(element, index, array) {

return 조건; // 조건이 true일 때 요소를 새 배열에 포함
});

element: 배열에서 현재 처리 중인 요소입니다.

index: 배열에서 현재 처리 중인 요소의 인덱스입니다.

array: filter()가 호출된 배열입니다.

  • 이 메서드는 주어진 함수 조건에 맞는 모든 요소를 새로운 배열로 반환합니다.
  • 콜백 함수를 인자로 받으며, 이 콜백 함수는 각 요소를 평가하여 'true' 또는 'false'를 반환합니다.
  • true를 반환하는 요소만이 새 배열에 포함됩니다.
  • filter()는 원본 배열을 변경하지 않으며, 조건에 맞는 요소들로 구성된 새로운 배열을 반환합니다.
{

}

위의 예시에서는 숫자 배열에서 짝수만 필터링하여 새로운 배열을 생성하는 방법을 보여줍니다.
이 코드는 배열 numbers의 각 요소를 콜백 함수에 전달하고, 짝수인지 확인합니다.
짝수일 경우 true를 반환하며, 이 때문에 해당 요소는 evenNumbers 배열에 포함됩니다.

16. 💛💛💛map()

const newArray = 배열.map(function(element, index, array){

return 새로운 값;
});

  • 이 메서드는 주어진 함수를 호출하여 각 요소에 대한 결과를 모아 새로운 배열을 만듭니다.
  • 이 메서드는 원본 배열의 모든 요소를 하나씩 처리하여 그 결과를 새 배열의 해당 위치에 저장합니다.
  • map()은 배열의 길이를 변경하지 않고, 원본 배열은 수정하지 않으면서 각 요소에 대해 주어진 함수를 실행한 결과로 구성된 새 배열을 반환합니다.
  • map() 메서드는 데이터를 변환하거나 수정하여 새로운 형태의 데이터를 생성할 때 매우 유용합니다.
  • 예를 들어, 객체 배열에서 특정 키의 값을 수정하거나 새로운 키를 추가하는 등의 작업에 자주 사용됩니다.
{
const numbers = [1, 2, 3, 4, 5];
const squaredNumbers = numbers.map(function(number) {
    return number * number; // 각 숫자를 제곱
});
console.log(squaredNumbers); // 출력: [1, 4, 9, 16, 25]
}

위의 예시에서는 숫자 배열의 각 요소를 제곱하여 새로운 배열을 생성하는 방법을 보여줍니다.
이 코드는 numbers 배열의 각 요소에 대해 제곱 연산을 수행하고, 그 결과를 squaredNumbers라는 새 배열에 저장합니다. 결과적으로, 각 숫자의 제곱값을 요소로 하는 새 배열이 출력됩니다.

21. 💛💛join()

const newString = 배열.join(구분자);

💡 구분자 : 배열의 각 요소를 연결할 때 사이에 끼워넣을 문자열입니다. 이 인자는 선택적이며, 생략하면 요소들 사이에 쉼표가 들어갑니다.

  • 메서드는 배열의 요소를 연결하여 하나의 문자열로 반환합니다.
  • 이 메서드는 배열 요소 사이에 삽입할 문자열을 인자로 받으며, 주어지지 않은 경우 기본적으로 쉼표(,)를 사용합니다.
  • join()은 원본 배열을 변경하지 않고, 요소들을 연결한 새로운 문자열을 반환합니다.
  • 'join()' 메서드는 배열의 요소들을 하나의 문자열로 결합할 때 매우 유용합니다.
  • 이를 통해 간단하게 배열 데이터를 텍스트 형태로 출력하거나 파일로 저장할 데이터를 준비할 수 있습니다.
  • 사용자 정의 구분자를 설정함으로써 다양한 형식의 데이터를 생성할 수 있으며, 이는 로그 파일 생성이나 데이터 전송 등에서 특히 유용하게 사용될 수 있습니다.
{
// 01. 간단한 배열을 문자열로 변환:
const fruits = ['apple', 'banana', 'mango'];
const resultString = fruits.join(); // 구분자를 지정하지 않았기 때문에 쉼표가 사용됨
console.log(resultString); // 출력: "apple,banana,mango"

// 02. 쉼표 대신 다른 구분자 사용:
const fruits = ['apple', 'banana', 'mango'];
const resultString = fruits.join(' - '); // 구분자로 하이픈과 공백 사용
console.log(resultString); // 출력: "apple - banana - mango"

// 03. 빈 문자열을 구분자로 사용:
const letters = ['H', 'e', 'l', 'l', 'o'];
const word = letters.join(''); // 구분자로 빈 문자열 사용
console.log(word); // 출력: "Hello"

}

24. 💛reduce()

from functools import reduce
def function(accumulator, element):
return new_accumulator_value
result = reduce(function, sequence, initial_value)

📍 function: 두 개의 인자를 받는 함수로, 첫 번째 인자는 누적 값(accumulator)이고, 두 번째 인자는 현재 시퀀스의 원소입니다.
이 함수의 반환 값은 다음 누적 값으로 사용됩니다.

📍 sequence: 이 함수가 적용될 시퀀스 (리스트, 튜플 등).

📍 initial_value: (선택적) 첫 번째 누적 값으로 사용할 초기 값입니다. 이 값을 생략하면 시퀀스의 첫 번째 원소가 초기 누적 값으로 사용됩니다.

  • 메서드는 배열의 각 요소에 주어진 리듀서 함수를 실행하고, 하나의 결과값을 반환합니다.
  • 이 함수는 두 개의 인자를 받는 함수와 시퀀스를 인자로 받아, 시퀀스의 원소를 왼쪽에서 오른쪽으로 함수를 적용하며 단일값으로 축소(reduce)합니다.
{
// 01. 모든 원소 합하기:
from functools import reduce

numbers = [1, 2, 3, 4, 5]
sum_result = reduce(lambda x, y: x + y, numbers)
print(sum_result)  # 출력: 15

// 02. 리스트의 원소들을 곱하는 예제:
from functools import reduce

numbers = [1, 2, 3, 4, 5]
result = reduce(lambda x, y: x * y, numbers)
print(result)  # 출력: 120

// 03. 초기값을 사용하는 예제:
from functools import reduce

numbers = [1, 2, 3, 4, 5]
result = reduce(lambda x, y: x + y, numbers, 10)
print(result)  # 출력: 25
}

26. 💛reverse()

__.reverse()

  • 메서드는 배열 요소의 순서를 반대로 하고, 그 결과로 변경된 배열을 반환합니다.
  • 이 메소드는 리스트 내의 원소들의 순서를 변경하여, 첫 번째 원소가 마지막이 되고 마지막 원소가 첫 번째가 되도록 합니다.
  • reverse()는 원본 리스트를 직접 변경하므로 반환값은 None입니다.
  • reverse()는 리스트를 뒤집을 때 매우 유용하며, 특히 데이터 처리나 알고리즘 구현에서 순서를 역전시켜야 할 경우 자주 사용됩니다.
  • reverse()는 원본 리스트를 직접 변경하는 점을 유의해야 합니다. 
{

words = ['Hello', 'world', 'Python', 'is', 'fun']

numbers = [10, 20, 30, 40, 50]

matrix = [[1, 2], [3, 4], [5, 6]]
matrix.reverse()

}

29. 💛💛slice()

__.slice(시작인덱스, 종료인덱스)

  • 메서드는 배열의 지정된 부분을 새로운 배열로 반환합니다.
  • JavaScript의 slice() 메서드는 배열에서 특정 부분을 새로운 배열로 복사해 생성하는데 사용됩니다.
  • 원본 배열은 변경되지 않습니다.
  • 이 메서드는 두 개의 매개변수를 받는데, 시작 인덱스와 종료 인덱스입니다.
{
// 01. 기본 예제
let array = [1, 2, 3, 4, 5];
let newArray = array.slice(1, 3); //      [2, 3]

// 02. 음수 인덱스: : 음수 인덱스를 사용하면 배열의 끝에서부터 요소를 세어 갑니다.
// 예를 들어, -1은 배열의 마지막 요소를 가리킵니다.
let array = [1, 2, 3, 4, 5];
let newArray = array.slice(-3, -1); //    [3, 4]

// 03. 인덱스 생략 : 인덱스 생략: 시작 인덱스만 제공하면 해당 인덱스부터 배열 끝까지 모든 요소를 포함한 새 배열을 생성합니다.
let array = [1, 2, 3, 4, 5];
let newArray = array.slice(2); // [3, 4, 5]

// 04. 전체 배열 복사: 시작 인덱스와 종료 인덱스를 모두 생략하면 전체 배열의 복사본을 생성합니다.
let array = [1, 2, 3, 4, 5];
let newArray = array.slice(); // [1, 2, 3, 4, 5]
}

📍 시작 인덱스 (start): 복사를 시작할 배열의 인덱스. 이 인덱스를 포함하여 복사가 시작됩니다. 매개변수를 지정하지 않으면 배열의 처음부터 복사를 시작합니다.
📍 종료 인덱스 (end): 복사를 끝낼 배열의 인덱스. 이 인덱스는 복사에 포함되지 않습니다. 즉, end-1 인덱스까지만 복사됩니다. 이 매개변수는 선택적입니다. 지정하지 않을 경우 배열의 끝까지 복사합니다.

결과 확인하기
[2, 3]
[3, 4]
[3, 4, 5]
[1, 2, 3, 4, 5]

30. 💛💛splice()

array.splice(start, deleteCount, item1, item2, ..., itemN)

start: 배열에서 수정을 시작할 인덱스입니다.
deleteCount: 배열에서 제거할 요소의 수입니다. 이 값이 0이면 아무 요소도 제거되지 않습니다. 이 값을 생략하면 start 인덱스부터 모든 요소가 제거됩니다.
item1, item2, ..., itemN: 배열에 추가할 요소들입니다. 이 인자들은 선택적이며, 제공되는 경우 start 인덱스 위치에 삽입됩니다.

  • 메서드는 배열에서 하나 이상의 요소를 추가하거나 교체하거나, 제거하고, 변경된 배열을 반환합니다.
  • 이 메소드는 매우 유연하며, 배열의 구조를 동적으로 수정할 때 매우 유용합니다.
  • splice()는 배열을 직접 수정합니다. 이는 메소드 호출 결과로 원본 배열이 변경된다는 것을 의미합니다.
  • 배열 요소를 제거하거나 추가할 때 사용하면 배열 내부의 데이터를 유연하게 관리할 수 있습니다.
  • 동적인 데이터 구조가 필요할 때 또는 배열의 특정 부분을 다른 데이터로 교체해야 할 때 매우 효과적입니다.
{
// 01. 요소 삭제하기 : 배열에서 특정 요소를 삭제하려면 start와 deleteCount를 지정합니다.
let array = [1, 2, 3, 4, 5];
array.splice(2, 1); // 인덱스 2에서 요소 하나(3)를 삭제
console.log(array); // [1, 2, 4, 5]

let fruits = ["apple", "banana", "cherry", "date"];
fruits.splice(1, 2); // "banana"와 "cherry" 삭제
console.log(fruits); // ["apple", "date"]

// 02. 요소 교체하기 : 배열의 특정 요소를 다른 요소로 교체하려면, 삭제할 요소의 수와 새로 추가할 요소들을 지정합니다.
let numbers = [1, 2, 3, 4, 5];
numbers.splice(2, 1, 99); // 인덱스 2의 요소(3)를 99로 교체
console.log(numbers); // [1, 2, 99, 4, 5]

// 03. 요소 추가하기 : 배열에 새 요소를 추가하려면, deleteCount를 0으로 설정하고, 추가할 요소들을 넣습니다.
let colors = ["red", "blue", "green"];
colors.splice(1, 0, "yellow", "pink"); // 인덱스 1에 "yellow"와 "pink" 추가
console.log(colors); // ["red", "yellow", "pink", "blue", "green"]
}

07. 💛💛sort()

array.sort(compareFunction)

compareFunction (선택적): 요소 정렬 방법을 정의하는 함수입니다. 이 함수를 지정하지 않으면 요소들은 문자열로 변환되어 유니코드 순서대로 정렬됩니다.

  • 메서드는 배열의 요소를 적절한 위치에 정렬하고, 그 배열을 반환합니다.
  • sort() 메서드는 배열의 요소를 정렬하는데 사용됩니다.
  • 이 메서드는 배열 내의 요소들을 문자열로 변환하고, 유니코드 값에 따라 사전 순으로 정렬합니다.
  • sort()는 배열을 직접 수정합니다.
{
// 01. 문자열 배열 정렬:
let fruits = ["Banana", "Orange", "Apple", "Mango"];
fruits.sort();
console.log(fruits); // ["Apple", "Banana", "Mango", "Orange"]

// 02. 숫자 배열 정렬: 숫자를 올바르게 정렬하기 위해 compareFunction을 사용합니다.
let numbers = [40, 100, 1, 5, 25];
numbers.sort((a, b) => a - b);
console.log(numbers); // [1, 5, 25, 40, 100]

// 이 경우,a - b가 양수, 음수, 또는 0이 되는 것에 따라 배열이 오름차순으로 정렬됩니다.

// 03. 내림차순 정렬 : 내림차순 정렬을 하려면 b - a를 사용합니다.
let numbers = [40, 100, 1, 5, 25];
numbers.sort((a, b) => b - a);
console.log(numbers); // [100, 40, 25, 5, 1]

}
