{
    let x = 100, y = 200, z = "javascript";

    console.log(x, y, z);
}

    console.log(x, y, z);
}

03. 배열 : 데이터 불러오기

배열에 저장된 데이터를 읽어오는 것을 말합니다.
배열은 여러 개의 데이터를 ㄹ저장하는 공간입니다. 

{
    const arr = [100, 200, "javascript"];

    console.log("03.배열 : 데이터 불러오기");
    console.log(arr[0], arr[1], arr[2]);
}

04.배열 : 데이터 불러오기: 갯수

배열에 저장된 데이터의 갯수를 읽어오는 것을 말합니다.
배열의 갯수 데이터를 불러오는 방법은 다음과 같습니다.
- 배열의 크기를 반환하는 내장 함수인 length를 사용합니다. 

{
    const arr = [100, 200, "javascript"];

    console.log("04.배열 : 데이터 불러오기: 갯수");
    console.log(arr.length);    // 3
}
결과 확인하기
3

06.배열 : 데이터 불러오기: forEach()문

배열의 forEach문은 배열의 각 요소에 대해 함수를 호출하는 메서드입니다.
함수를 호출할 때마다 배열의 다으 요소로 이동하며, 모든 요소를 처리할 때까지 반복합니다. 

{
    const arr = [100, 200, 300, 400, 500, 600, 700, 800, 900];

    console.log("06.배열 : 데이터 불러오기: forEach()문");
    // 01
    arr.forEach(function (element) {
        console.log(element);
    });
    // 02
    arr.forEach(function (element, index) {
        console.log(index);
    });
    //03
    arr.forEach(function (element, index, array) {
        console.log(array);
    });
}

07.배열 : 데이터 불러오기: for of

배열의 for of문은 배열의 각 요소를 순회하면서 각 요소에 접근할 수 있는 방법을 제공하는 구문입니다.
for of문은 ES6에서 새롭게 추가된 구문으로, 기존의 for문보다 더 간결하고 직관적으로 배열의 요소를 순회할 수 있습니다. 

{
    const arr = [100, 200, 300, 400, 500, 600, 700, 800, 900];

    console.log("07.배열 : 데이터 불러오기: for of");
    for (let element of arr) {
        console.log(element);
    }
}

08. 배열 : 데이터 불러오기 : for in

배열의 for in문은 배열의 각 요소에 대해 인덱스를 순회하면서 각 요소에 접근할 수 있는 방법을 제공하는 구문입니다.
for in문은 객체의 속성을 순회하는 데에도 사용됩니다. 

{
    const arr = [100, 200, 300, 400, 500, 600, 700, 800, 900];

    console.log("08.배열 : 데이터 불러오기: for in");
    for (let element in arr) {
        console.log(arr[element]);
    }
}

09. 배열 : 데이터 불러오기: map()
